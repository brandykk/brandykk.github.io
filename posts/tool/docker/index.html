Docker下载

# archlinux
sudo pacman -Syu docker
# 将用户lv追加进docker用户组
# 将当前用户添加进docker用户组 -a append; -G group
sudo usermod -a -G docker lv

Docker运行

# 修改docker默认镜像安装路径(没有则新建)
vim /etc/docker/daemon.json
# 修改文件中的路径
{
    "registry-mirrors":
        [
            "https://registry.cn-hangzhou.aliyuncs.com",
            "https://hub-mirror.c.163.com"

        ]
} 
# 重新加载配置文件
systemctl daemon-reload
# 启动docker服务
systemctl start/stop docker.service

Docker基础使用

# 下载镜像
docker pull archlinux
# 查看镜像
docker images 

# 创建容器
docker run
--name mariadb //容器名称
-p 13306:3306 //将容器端口映射到服务器：第一个参数是主机端口，第二个是容器端口
-e MYSQL_ROOT_PASSWORD=xxxxx //xxxxx ：设置为自己的密码
-v /data/mariadb/data:/var/lib/mysql // data/mariadb/data即二步骤中新建挂载数据的目录
-d mariadb:10.1.21 //如果你的mariadb是指定版本，那这块启动时需要携带指定，否则服务器会重新下载最新版本的容器 

# 启动/停止容器
docker start/stop <容器ID或容器名>
# 删除镜像之前需要先删除对应的容器
# 查看容器
docker ps -a
# 删除容器
docker rm 容器id
# 查看docker image
docker images
# 删除image
docker rmi 镜像id

mariadb

# 创建需要映射的目录
mkdir /mnt/project/archLinux/docker/mariadb/data/

# 拉取mariadb镜像
docker pull mariadb

# 创建容器,并指定密码和端口
docker run --name docker-mariadb -p 13306:3306 -e MYSQL_ROOT_PASSWORD=root -v /mnt/project/archLinux/docker/mariadb/data/:/var/lib/mysql mariadb

# 启动容器
docker start docker-mariadb
  • –name mariadb //容器名称
  • -p 13306:3306 //将容器端口映射到服务器：第一个参数是主机端口，第二个是容器端口
  • -e MYSQL_ROOT_PASSWORD=xxxxx //xxxxx ：设置为自己的密码
  • -v /data/mariadb/data:/var/lib/mysql // data/mariadb/data即二步骤中新建挂载数据的目录
  • -d mariadb:10.1.21 //如果你的mariadb是指定版本，那这块启动时需要携带指定，否则服务器会重新下载最新版本的容器

redis

# 创建映射目录
mkdir -p /mnt/project/archLinux/docker/redis/{data,etc}

# 拉取redis镜像
docker pull redis

# 创建redis.conf放入创建好的etc文件夹下

# 创建容器
docker run --name docker-redis  -p 16379:6379 -v 
/mnt/project/archLinux/docker/redis/data/:/data -v /mnt/project/archLinux/docker/redis/etc/redis.conf:/etc/redis/redis.conf -d redis redis-server /etc/redis/redis.conf --appendonly yes
  • -p 16380:6380 端口映射：前表示主机部分，：后表示容器部分。
  • –name myredis 指定该容器名称，查看和进行操作都比较方便。
  • -v 挂载目录，规则与端口映射相同。
  • -d redis 表示后台启动redis
  • redis-server /etc/redis/redis.conf 以配置文件启动redis，加载容器内的conf文件，最终找到的是挂载的目录/usr/local/docker/redis.conf
  • appendonly yes 开启redis 持久化
