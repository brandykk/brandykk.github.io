查询模式sql

SELECT * FROM user_tables ORDER BY table_name;

-- 获取当前用户下的模式名
SELECT distinct object_name FROM all_objects WHERE object_type='SCH';

-- 获取当前用户下模式（库）的信息
SELECT * FROM dba_segments;

-- 获取模式下的表
SELECT table_name FROM dba_tables WHERE owner='模式（库）名';
